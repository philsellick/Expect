Namespace: MM

MM

MM is the primary interface to all MindMeld JavaScript SDK functionality. Call MM#init before anything else. Next obtain a token via MM#getToken to start making API calls.

callApi(method, path, params, success, error)

Makes a call directly to the MindMeld API. This method can be used to make calls to any path of the MindMeld API that are not part of the namespaces

Name Type Description
method string

HTTP method to use for API call ('GET', 'POST', or 'DELETE')
path string

API endpoint path (e.g., 'session/:sessionid/textentries')
params QueryParameters optional

Parameters to be sent to MindMeld API. Params are URL encoded for GET and DELETE requests and are sent as POST data for POST requests
success APISuccessCallback optional

A callback function to be called if the API request succeeds. The function receives one argument containing the data returned from the server
error qAPIErrorCallback optional

A callback function to be called if the API request fails. The function receives one argument, the error message returned from the server
Examples

Example GET request to the session text entries endpoint 

 function callAPI () {
    MM.callApi('GET', 'session/47978/textentries', null, onGetTextEntries);
 }
 function onGetTextEntries (response) {
    var responseData = response.data;
 }

Example POST request to the application's publish event endpoint 

 function callAPI () {
    var eventData = {
        name: 'custom event name',
        payload: 'test payload'
    };
    MM.callApi('POST', 'events', eventData, onPublishEvent);
 }
 function onPublishEvent (response) {
    var responseData = response.data;
 }

get(params, onSuccess, onFail)

Get information about the application. User privileges allow access to basic application information. Admin privileges allow access to extended information about the application. Note that, if an onUpdate handler has already been specified for this object, the onUpdate handler will be invoked first, followed by any specified 'onSuccess' callback.

Name Type Description
params QueryParameters optional

query parameters when fetching the application object
onSuccess APISuccessCallback optional

callback for when getting application data was successful
onFail APIErrorCallback optional

callback for when getting application data failed
Example
 function getApplicationInfo () {
    MM.get(null, onGetApplicationInfo);
 }
 function onGetApplicationInfo (response) {
    var applicationInfo = response.data;
 }

getToken(credentials, onSuccess, onError)

Requests a new admin or user token from the API and stores it locally. This token is automatically used for all subsequent requests to the API. If we successfully obtain a token, MM#getToken automatically calls MM#setActiveUserID with the appropriate user id

Name Type Description
credentials Object

credentials for obtaining an API token. Please refer to documentation here for details
onSuccess function optional

callback for when token obtained successfully
onError function optional

callback for when there was an error obtaining token
Examples

Example code to get a token 

 var credentials = {...}; // admin credentials, simple user credentials, or user credentials
 MM.getToken(credentials, onGetToken);

 function onGetToken (result) {
    var token = result.token;
 }

Example credentials to get an admin token 

 var adminCredentials = {
    appsceret: '<appsecret>'
 };

Example credentials to get a simple user token 

 var simpleUserCredentials = {
    appsceret: '<appsecret>',
    simple: {
        userid: 'einstein79',
        name: 'Albert Einstein'
    }
 }

Example credentials to get a user token 

 var fbUserId = 'Facebook User Id';
 var fbAuthToken = 'Facebook User Token';
 var userCredentials = {
     facebook: {
        userid: fbUserId,
        token: fbAuthToken
     }
 };

json(){Object}

Helper function returns the JSON data for the current application. You must have called MM#get first, before MM#json returns any data.

Example
 function getApplicationInfo () {
    MM.get(null, onGetApplicationInfo);
 }
 function onGetApplicationInfo () {
    var applicationInfo = MM.json();
    // MM.json() returns a JSON object containing data received from MM.get()
 }

onUpdate(updateHandler)

Sets the app object's onUpdate handler. Pass null as the updateHandler parameter to deregister a previously set updateHandler. If the updateHandler has been set, it is automatically called when application info is fetched (e.g. MM#get)

Name Type Description
updateHandler APISuccessCallback optional

callback for when the app object updates
Example
 function getApplicationInfo () {
    MM.onUpdate(onGetApplicationInfo); // Set the updateHandler
    MM.get(); // Fetch application info
 }
 function onGetApplicationInfo (response) {
    var applicationInfo = response.data;
 }

publish(event, payload)

Publish a new, custom event on the app channel

Name Type Description

event name
payload EventPayload optional

payload for event
Example

Code snippet to subscribe and publish a custom event on the application channel 

 function publishEvent() {
    // First subscribe to an event. In this case we are
    // subscribing to an event named 'testEvent'
    MM.subscribe('testEvent', onTestEvent, onTestEventSubscribed);
 }
 function onTestEventSubscribed () {
    console.log('Successfully subscribed to testEvent on application channel');
    // Now that we have successfully subscribed to the 'testEvent' event,
    // publish a 'testEvent' with the payload containing the string
    // 'custom payload'
    MM.publish('testEvent', 'custom payload');

 function onTestEvent (payload) {
    // the payload parameter is 'custom payload'
    console.log('Received testEvent with payload: ' + payload);
 }

revokeToken(onSuccess, onError)

Revokes the current API token. Note that subsequent calls to the MindMeld API will not work until a new token is obtained

Name Type Description
onSuccess function optional

callback for when token is successfully revoked
onError function optional

callback for when there was an error revoking token
Example
 function revokeTokenExample () {
    // First, get a token. In this example, we are getting an admin token
    var credentials = {
        appsecret: '<appsecret>'
    };
    MM.getToken(credentials, onGetToken);
 }
 function onGetToken () {
    // Now that we have a token, try an API request
    MM.get(null, onGetApplicationInfo);
 }
 function onGetApplicationInfo (result) {
    // Request succeeds because we have a token
    var applicationInfo = result.data;
    // Now, let's revoke the token
    MM.revokeToken (onRevokeToken);
 }
 function onRevokeToken () {
    // Now that we have revoked the token, try an API Request
    MM.get(null, onGetApplicationInfo, onGetApplicationError);
 }
 function onGetApplicationError (error) {
    console.log('Call failed. Error code ' + error.code + ': ' + error.message);
    // "Call failed. Error code 8: No token parameter was included in the api request"
 }

setActiveSessionID(sessionid, onSuccess, onError)

Sets the active session to a specified session id. MM#setActiveSessionID also tries to fetch the session object and clears all event handlers from the previous session. You must call setActiveSessionID before calling any of the functions in the MM.activeSession namespace

Name Type Description
sessionid string

session id to set active session to
onSuccess APISuccessCallback optional

callback for when session data was successfully fetched
onError APIErrorCallback optional

callback for when there was an error fetching session data
Example
 function testSetActiveSessionID () {
    MM.setActiveSessionID('<session id>');
 }

setActiveUserID(userid, onSuccess, onError)

Sets the active user to a specified user id. MM#setActiveUserID also tries to fetch the user object and clears all event handlers from the previous user. MM#setActiveUserID is automatically called after successfully calling MM#getToken. You should only to call this method if you are using an admin token and want to impersonate other users, or if you call MM#setToken with an existing token and already know the corresponding user id

Name Type Description
userid string
onSuccess APISuccessCallback optional

callback for when user data successfully fetched
onError APIErrorCallback optional

callback for when there was an error fetching user data
Example
 var userToken = '<known user token>';
 MM.setToken(userToken, onTokenValid);

 function onTokenValid () {
    MM.setActiveUserID('<known mindmeld user  id>', onGetUserInfo);
 }
 function onGetUserInfo (response) {
    var userInfo = response.data;
 }

setToken(token, onTokenValid, onTokenInvalid)

Set the MM token directly instead of calling MM#getToken. This function also provides valid/invalid callbacks to determine if the given token is valid or not. Regardless of the token being valid, MM#setToken always sets the token used by MM. Unlike MM#getToken, MM#setToken does not automatically call MM#setActiveUserID

Name Type Description
token string

token to be used by SDK
onTokenValid function optional

callback for when given token is valid
onTokenInvalid function optional

callback for when given token is invalid
Example
 function setToken () {
    MM.setToken('<token>', onTokenValid, onTokenInvalid);
 }
 function onTokenValid () {
    // token is valid
 }
 function onTokenInvalid () {
    // token is invalid
 }

Subscribes to every event on the app channel

Name Type Description

callback for when an event on the app channel is fired
onSuccess function optional

callback for when subscription is successful
onError function optional

callback for when there is an error subscribing
Example
 function subscribeAllExample () {
    MM.subscribeAll(onApplicationChannelEvent, onSubscribeApplicationChannel);
 }

    MM.publish('eventA', 'payloadA');
    MM.publish('eventB', 'payloadB');
 }
 function onApplicationChannelEvent (eventObject) {
    var eventName = eventObject.event;
    var eventPayload = eventObject.payload;
    console.log('Received event ' + eventName +
        ' with payload ' + eventPayload);
    // Received event eventA with payload payloadA
    // Received event eventB with payload payloadB
 }

unsubscribe(eventName)

Unsubscribe from a custom event on the app channel

Name Type Description
eventName string

name of event to subscribe from
Example
 function unsubscribeExample() {
    // First subscribe to an event. In this case we are
    // subscribing to an event named 'testEvent'
    MM.subscribe('testEvent', onTestEvent, onTestEventSubscribed);
 }
 function onTestEventSubscribed () {
    console.log('Successfully subscribed to testEvent on application channel');
    // Now that we have successfully subscribed to the 'testEvent' event,
    // publish a 'testEvent'
    MM.publish('testEvent');
 }
 function onTestEvent (payload) {
    // onTestEvent will be called once after 'testEvent' is published
    console.log('received test event');
    // Now unsubscribe from 'testEvent'
    MM.unsubscribe('testEvent');
    // Publish 'testEvent' again
    MM.publish('testEvent');
    // Since we unsubscribed, onTestEvent won't be called anymore
 }

unsubscribeAll()

Unsubscribe from all events on the app channel

Example
 function unsubscribeAllExample () {
    // First subscribe to all events on app channel
    MM.subscribeAll(onApplicationEvent, onSubscribeApplicationChannel);
 }
 function onSubscribeApplicationChannel () {
    // publish the event 'testEvent'
    MM.publish('testEvent');
 }
 function onApplicationEvent (eventObject) {
    var eventName = eventObject.event;
    console.log('Received event ' + eventName);
    // Now unsubscribe from application events
    MM.unsubscribeAll();
    MM.publish('testEvent');
    // onApplicationEvent won't be called because we are unsubscribed
    // from all application level events
 }
